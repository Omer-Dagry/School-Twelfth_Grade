            cmd (30)        len username (2)            username     len email (15)          email  password
    f"{'signup'.ljust(30)}{str(len(username)).ljust(2)}{username}{str(len(email)).ljust(15)}{email}{password}"

    f"{'confirmation_code'.ljust(30)"

    client -> server
    f"{'confirmation_code'.ljust(30)}{input('Please enter your confirmation code (sent to your email): ')}"

    server -> client
    f"{"signup".ljust(30)}{status.ljust(6)}{reason}"
